**Step 2 — Create `packages/shared/src/utils/logger.ts`:**

New file. Pino logger with child logger factory for route-level context injection.

```typescript
import pino from "pino";

export const logger = pino({
  level: process.env.LOG_LEVEL || "info",
  ...(process.env.NODE_ENV !== "production" && {
    transport: {
      target: "pino-pretty",
      options: { colorize: true },
    },
  }),
});

export type Logger = pino.Logger;

export function createRouteLogger(
  route: string,
  context?: {
    correlation_id?: string;
    org_id?: string;
    session_id?: string;
  }
) {
  return logger.child({ route, ...context });
}
```

Key points:
- Production outputs raw JSON (no transport). Dev uses pino-pretty (installed in apps/admin, available at runtime).
- `createRouteLogger` produces a child with route name + optional correlation/org/session IDs baked in.
- Every subsequent `log.info(...)` or `log.error(...)` call on the child automatically includes these fields.

**Step 3 — Replace `packages/shared/src/utils/rate-limiter.ts`:**

Complete rewrite. Replace the entire in-memory implementation with Upstash Redis-backed version. Preserve the `RATE_LIMITS` constant and `getRateLimitHeaders` helper signature for backward compatibility.

```typescript
import { Ratelimit } from "@upstash/ratelimit";
import { Redis } from "@upstash/redis";

// Lazy init to avoid boot failures when env vars missing (build time, tests)
let redis: Redis | null = null;
function getRedis(): Redis {
  if (!redis) {
    redis = Redis.fromEnv();
  }
  return redis;
}

export const RATE_LIMITS = {
  webhook: { requests: 100, window: "60 s" as const },
  api:     { requests: 60,  window: "60 s" as const },
  widget:  { requests: 60,  window: "60 s" as const },
  auth:    { requests: 10,  window: "60 s" as const },
  signup:  { requests: 10,  window: "60 s" as const },
  cron:    { requests: 5,   window: "60 s" as const },
} as const;

export type RateLimitTier = keyof typeof RATE_LIMITS;

const limiters = new Map<string, Ratelimit>();

export function getRateLimiter(tier: RateLimitTier): Ratelimit {
  if (!limiters.has(tier)) {
    const config = RATE_LIMITS[tier];
    limiters.set(
      tier,
      new Ratelimit({
        redis: getRedis(),
        limiter: Ratelimit.slidingWindow(config.requests, config.window),
        prefix: `leadrwizard:ratelimit:${tier}`,
        analytics: true,
      })
    );
  }
  return limiters.get(tier)!;
}

export function getTierForPath(pathname: string): RateLimitTier {
  if (pathname.startsWith("/api/webhooks/")) return "webhook";
  if (pathname.startsWith("/api/widget/")) return "widget";
  if (pathname.startsWith("/api/signup/")) return "signup";
  if (pathname.startsWith("/api/cron/")) return "cron";
  return "api";
}

export function getRateLimitHeaders(result: { remaining: number; reset: number }): Record<string, string> {
  return {
    "X-RateLimit-Remaining": String(result.remaining),
    "X-RateLimit-Reset": String(Math.ceil(result.reset / 1000)),
  };
}
```

Key points:
- Lazy Redis init prevents crashes during `next build` when env vars are absent.
- Separate Ratelimit instance per tier so limits are independent.
- `slidingWindow` algorithm for smooth rate limiting (not fixed window).
- `RATE_LIMITS` includes `widget` and `signup` tiers (new, from research).
- `getRateLimitHeaders` signature updated to accept Upstash's `{ remaining, reset }` shape instead of the old custom `RateLimitResult`.

**Step 6 — Create `apps/admin/src/app/global-error.tsx`:**

React error boundary for the App Router root:

```typescript
"use client";

import * as Sentry from "@sentry/nextjs";
import { useEffect } from "react";

export default function GlobalError({
  error,
  reset,
}: {
  error: Error & { digest?: string };
  reset: () => void;
}) {
  useEffect(() => {
    Sentry.captureException(error);
  }, [error]);

  return (
    <html>
      <body>
        <div style={{ padding: "2rem", textAlign: "center" }}>
          <h2>Something went wrong</h2>
          <p style={{ color: "#666", marginTop: "0.5rem" }}>
            An unexpected error occurred. Our team has been notified.
          </p>
          <button
            onClick={reset}
            style={{
              marginTop: "1rem",
              padding: "0.5rem 1rem",
              cursor: "pointer",
              borderRadius: "0.375rem",
              border: "1px solid #d1d5db",
              backgroundColor: "#fff",
            }}
          >
            Try again
          </button>
        </div>
      </body>
    </html>
  );
}
```

Uses `"use client"` directive (required for error boundaries). Captures the error in Sentry on mount. Provides a reset button.
